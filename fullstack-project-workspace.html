// Example: TaskList component with AI Summarize button import { useState, useEffect } from 'react'; import axios from 'axios'; const API = import.meta.env.VITE_API_URL || 'http://localhost:5000/api'; export default function TaskList() { const [tasks, setTasks] = useState([]); const [summary, setSummary] = useState(''); const [loading, setLoading] = useState(false); useEffect(() => { const token = localStorage.getItem('token'); axios.get(`${API}/tasks`, { headers: { Authorization: `Bearer ${token}` } }).then(r => setTasks(r.data)); }, []); const summarizeTasks = async () => { setLoading(true); const res = await axios.post(`${API}/ai/summarize`, { tasks }); setSummary(res.data.summary); setLoading(false); }; return ( <div className="p-6"> {tasks.map(t => <TaskCard key={t._id} task={t} />)} <button onClick={summarizeTasks} disabled={loading}> {loading ? 'Summarizing...' : '🤖 Summarize with AI'} </button> {summary && <div className="ai-summary">{summary}</div>} </div> ); }

2Secured Express API with JWT

Node.js + Express + JWT

Build a RESTful Node/Express API following MVC architecture. Implement JWT authentication, protect all task routes, and add an AI summarization endpoint.

Requirements Checklist

  • POST /api/auth/register — hash password with bcrypt, return JWT
  • POST /api/auth/login — verify bcrypt hash, return JWT
  • Auth middleware: verifies Bearer token on protected routes
  • GET / POST / PUT / DELETE /api/tasks — full CRUD, JWT-protected
  • POST /api/ai/summarize — accepts tasks array, calls OpenAI/Gemini, returns summary
  • Input validation with express-validator
  • Global error handler middleware
  • CORS configured for frontend origin

Starter Code Reference

// routes/ai.js — AI summarization endpoint const express = require('express'); const router = express.Router(); const { OpenAI } = require('openai'); const authMiddleware = require('../middleware/auth'); const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY }); router.post('/summarize', authMiddleware, async (req, res) => { try { const { tasks } = req.body; if (!tasks || !tasks.length) { return res.status(400).json({ message: 'No tasks provided' }); } const taskList = tasks.map((t, i) => `${i+1}. [${t.completed?'Done':'Pending'}] ${t.title}` ).join('\n'); const completion = await openai.chat.completions.create({ model: 'gpt-4o-mini', messages: [{ role: 'user', content: `Summarize these tasks in 2-3 sentences:\n${taskList}` }] }); res.json({ summary: completion.choices[0].message.content }); } catch (err) { res.status(500).json({ message: 'AI service error', error: err.message }); } }); module.exports = router;

3MongoDB CRUD + AI Summarization

MongoDB + Mongoose + GenAI

Design the MongoDB schema, implement full CRUD operations, and complete the AI summarization integration end-to-end. Deploy to MongoDB Atlas.

Requirements Checklist

  • User schema: name, email (unique), password (hashed), timestamps
  • Task schema: title, description, completed (Boolean), priority, owner (ref: User), timestamps
  • CRUD: create task, list user's tasks, update task, delete task
  • Query filter: GET /api/tasks?completed=true
  • Index on { owner, completed } for query performance
  • AI summaries saved to DB (AISummary model with content + timestamp)
  • Deployed to MongoDB Atlas (free M0 cluster)

Starter Code Reference

// models/Task.js — Mongoose Schema const mongoose = require('mongoose'); const taskSchema = new mongoose.Schema({ title: { type: String, required: [true, 'Title is required'], trim: true, maxlength: 200 }, description: { type: String, default: '' }, completed: { type: Boolean, default: false }, priority: { type: String, enum: ['low', 'medium', 'high'], default: 'medium' }, owner: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true } }, { timestamps: true }); // Compound index for performance taskSchema.index({ owner: 1, completed: 1 }); module.exports = mongoose.model('Task', taskSchema);
